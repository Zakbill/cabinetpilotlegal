// Single string param (statut filter):
const [statut, setStatut] = useQueryState('statut', parseAsString.withDefault(''))

setStatut('')       // removes ?statut from URL

KPI reactivity approach:
Since DossiersTable is a client component with filter state, the cleanest approach is to:
1. Derive filtered dossiers INSIDE DossiersTable (using the nuqs state)
2. Pass the filtered array to the TanStack table
3. Pass the filtered array to KpiCards rendered INSIDE DossiersTable (not in page.tsx)
4. Remove KpiCards from page.tsx and render it inside DossiersTable instead
This keeps all filter logic co-located and avoids prop-drilling URL state up to the server component.

<div className="flex items-center gap-3 mb-4 flex-wrap">

  <select
    value={statut}
    onChange={(e) => setStatut(e.target.value || null)}
    className="text-sm border border-border rounded-md px-3 py-1.5 bg-background text-foreground focus:outline-none focus:ring-2 focus:ring-ring"

    <option value="En retard">En retard</option>
  </select>

  {/* Cabinet toggle buttons */}
  <div className="flex items-center gap-1.5 flex-wrap">
    {cabinetOptions.map(cabinet => (
      <button
        key={cabinet}
        onClick={() => {
          const isSelected = selectedCabinets.includes(cabinet)
          setSelectedCabinets(
            isSelected
              ? selectedCabinets.filter(c => c !== cabinet)
              : [...selectedCabinets, cabinet]
          )
        }}
        className={`text-xs px-2.5 py-1 rounded-full border transition-colors ${
          selectedCabinets.includes(cabinet)
            ? 'bg-primary text-primary-foreground border-primary'
            : 'bg-background text-muted-foreground border-border hover:border-foreground hover:text-foreground'
        }`}
      >
        {cabinet}
      </button>
    ))}
  </div>

  {/* Clear filters — only shown when filters are active */}
  {(statut || selectedCabinets.length > 0) && (
    <button
      onClick={() => { setStatut(null); setSelectedCabinets([]) }}
