```tsx
"use client"

import { motion } from 'framer-motion'
import { Card, CardContent } from '@/components/ui/card'
import type { Dossier } from '@/lib/mock-data'

interface KpiCardsProps {
  dossiers: Dossier[]
}

export function KpiCards({ dossiers }: KpiCardsProps) {
  const total = dossiers.length
  const nonCommence = dossiers.filter(d => d.statut === 'Non commencé').length
  const enCours = dossiers.filter(d => d.statut === 'En cours').length
  const termine = dossiers.filter(d => d.statut === 'Terminé').length
  const enRetard = dossiers.filter(d => d.statut === 'En retard').length

  const kpis = [
    { label: 'Total dossiers', value: total, className: 'text-foreground' },
    { label: 'Non commencé', value: nonCommence, className: 'text-foreground' },
    { label: 'En cours', value: enCours, className: 'text-foreground' },
    { label: 'En retard', value: enRetard, className: 'text-destructive' },
  ]

  return (
    <div className="grid grid-cols-2 gap-4 mb-8 lg:grid-cols-4">
      {kpis.map((kpi, i) => (
        <motion.div
          key={kpi.label}
          initial={{ opacity: 0, y: -4 }}
          animate={{ opacity: 1, y: 0 }}
          transition={{ duration: 0.3, delay: i * 0.05 }}
        >
          <Card className="bg-background/60 backdrop-blur-sm border-border">
            <CardContent className="p-6">
              <p className={`text-3xl font-semibold font-display ${kpi.className}`}>
                {kpi.value}
              </p>
              <p className="text-sm text-muted-foreground mt-1">{kpi.label}</p>
            </CardContent>
          </Card>
        </motion.div>
      ))}
    </div>
  )
}
```

Create src/components/dashboard/DossiersTable.tsx as a "use client" component.

This component receives `dossiers: Dossier[]` as props (passed from page.tsx), manages its own sorting state and selected-dossier state, and renders the TanStack table + Sheet.

```tsx
"use client"

import { useState } from 'react'
import {
  useReactTable,
  getCoreRowModel,
  getSortedRowModel,
  flexRender,
  type ColumnDef,
  type SortingState,
} from '@tanstack/react-table'
import { MoreHorizontal, ArrowUpDown } from 'lucide-react'
import { Avatar, AvatarFallback, AvatarGroup, AvatarGroupCount } from '@/components/ui/avatar'
import { DossierSheet } from './DossierSheet'
import { getUrgency, urgencyClass, urgencyRowClass } from '@/lib/date-utils'
import type { Dossier, Intervenant } from '@/lib/mock-data'

// Status badge color map — Tailwind utility classes per CONTEXT.md spec
const STATUS_STYLES: Record<Dossier['statut'], string> = {
  'Non commencé': 'bg-slate-100 text-slate-700',
  'En cours':     'bg-blue-50 text-blue-700',
  'Terminé':      'bg-green-50 text-green-700',
  'En retard':    'bg-red-50 text-red-600',
}

function StatusBadge({ statut }: { statut: Dossier['statut'] }) {
  return (
    <span className={`inline-flex items-center rounded-full px-2.5 py-0.5 text-xs font-medium ${STATUS_STYLES[statut]}`}>
      {statut}
    </span>
  )
}

function IntervenantsCell({ intervenants }: { intervenants: Intervenant[] }) {
  const visible = intervenants.slice(0, 3)
  const overflow = intervenants.length - visible.length
  return (
    <AvatarGroup>
      {visible.map(p => (
        <Avatar key={p.initials} size="sm">
          <AvatarFallback>{p.initials}</AvatarFallback>
        </Avatar>
      ))}
      {overflow > 0 && <AvatarGroupCount>+{overflow}</AvatarGroupCount>}
    </AvatarGroup>
  )
}

interface DossiersTableProps {
  dossiers: Dossier[]
}

export function DossiersTable({ dossiers }: DossiersTableProps) {
  const [sorting, setSorting] = useState<SortingState>([])
  const [selectedDossier, setSelectedDossier] = useState<Dossier | null>(null)

  const columns: ColumnDef<Dossier>[] = [
    {
      accessorKey: 'nom_societe',
      header: ({ column }) => (
        <button
          className="flex items-center gap-1 text-xs font-medium text-muted-foreground hover:text-foreground transition-colors"
          onClick={column.getToggleSortingHandler()}
        >
          Dossier <ArrowUpDown className="size-3" />
        </button>
      ),
      cell: ({ row }) => (
        <div className="flex items-center gap-3">
          <Avatar size="sm">
            <AvatarFallback>
              {row.original.nom_societe.slice(0, 2).toUpperCase()}
            </AvatarFallback>
          </Avatar>
          <div>
            <p className="font-medium text-foreground text-sm">{row.original.nom_societe}</p>
            <p className="text-xs text-muted-foreground">{row.original.cabinet}</p>
          </div>
        </div>
      ),
    },
    {
      accessorKey: 'forme_juridique',
      header: 'Forme juridique',
      cell: ({ getValue }) => (
        <span className="text-sm text-foreground">{getValue<string>()}</span>
      ),
    },
    {
      accessorKey: 'statut',
      header: 'Statut',
      cell: ({ getValue }) => <StatusBadge statut={getValue<Dossier['statut']>()} />,
    },
    {
      accessorKey: 'date_echeance',
      header: ({ column }) => (
        <button
          className="flex items-center gap-1 text-xs font-medium text-muted-foreground hover:text-foreground transition-colors"
          onClick={column.getToggleSortingHandler()}
        >
          Échéance <ArrowUpDown className="size-3" />
        </button>
      ),
      cell: ({ getValue }) => {
        const dateStr = getValue<string>()
        const urgency = getUrgency(dateStr)
        return (
          <span className={`text-sm ${urgencyClass[urgency]}`}>
            {new Date(dateStr).toLocaleDateString('fr-FR')}
          </span>
        )
      },
    },
    {
      accessorKey: 'intervenants',
      header: 'Intervenants',
      cell: ({ getValue }) => (
        <IntervenantsCell intervenants={getValue<Intervenant[]>()} />
      ),
      enableSorting: false,
    },
    {
      id: 'actions',
      header: '',
      cell: () => (
        <button
          className="text-muted-foreground hover:text-foreground p-1 rounded transition-colors"
          onClick={(e) => e.stopPropagation()}
          aria-label="Actions"
        >
          <MoreHorizontal className="size-4" />
        </button>
      ),
      enableSorting: false,
    },
  ]

  const table = useReactTable({
    data: dossiers,
    columns,
    state: { sorting },
    onSortingChange: setSorting,
    getCoreRowModel: getCoreRowModel(),
    getSortedRowModel: getSortedRowModel(),
  })

  return (
    <>
      <div className="rounded-lg border border-border bg-background/60 backdrop-blur-sm overflow-hidden">
        <table className="w-full">
          <thead>
            {table.getHeaderGroups().map(headerGroup => (
              <tr key={headerGroup.id} className="border-b border-border">
                {headerGroup.headers.map(header => (
                  <th
                    key={header.id}
                    className="px-4 py-3 text-left text-xs font-medium text-muted-foreground"
                  >
                    {header.isPlaceholder
                      ? null
                      : flexRender(header.column.columnDef.header, header.getContext())}
                  </th>
                ))}
              </tr>
            ))}
          </thead>
          <tbody className="divide-y divide-border">
            {table.getRowModel().rows.map(row => {
              const urgency = getUrgency(row.original.date_echeance)
              return (
                <tr
                  key={row.id}
                  className={`cursor-pointer hover:bg-muted/50 transition-colors ${urgencyRowClass[urgency]}`}
                  onClick={() => setSelectedDossier(row.original)}
                >
                  {row.getVisibleCells().map(cell => (
                    <td key={cell.id} className="px-4 py-4">
                      {flexRender(cell.column.columnDef.cell, cell.getContext())}
                    </td>
                  ))}
                </tr>
              )
            })}
          </tbody>
        </table>

        {dossiers.length === 0 && (
          <div className="py-12 text-center text-muted-foreground text-sm">
            Aucun dossier ne correspond aux filtres sélectionnés.
          </div>
        )}
      </div>

      <DossierSheet
        dossier={selectedDossier}
        onClose={() => setSelectedDossier(null)}
      />
    </>
  )
}
```
